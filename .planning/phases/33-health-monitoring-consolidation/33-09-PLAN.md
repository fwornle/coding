    ```js
    /**
     * Service liveness probe helpers — Phase 33 G1.
     * Used by scripts/health-coordinator.js to populate currentState.services
     * status from rule check_type ('http_health' or 'port_listening').
     *
     * Contract: every probe returns { status, latency_ms, error } where
     *   status = 'running' | 'stopped' | 'unknown'
     * SPEC R6: NEVER 'healthy' / NEVER default to 'running' on exception.
     *
     * @module lib/utils/service-probe
     */
    import net from 'node:net';

    /**
     * HTTP GET probe with timeout. Treats 2xx/3xx as 'running', other codes
     * as 'stopped', network errors as 'stopped', unparseable URLs as 'unknown'.
     * @param {string} endpoint
     * @param {number} timeoutMs
     * @returns {Promise<{status: string, latency_ms: number | null, error: string | null}>}
     */
    export async function probeHttpHealth(endpoint, timeoutMs = 3000) {
      if (!endpoint || typeof endpoint !== 'string') {
        return { status: 'unknown', latency_ms: null, error: 'invalid endpoint' };
      }
      let url;
      try { url = new URL(endpoint); } catch (err) {
        return { status: 'unknown', latency_ms: null, error: `bad URL: ${err.message}` };
      }
      const ctrl = new AbortController();
      const t = setTimeout(() => ctrl.abort(), timeoutMs);
      const t0 = Date.now();
      try {
        const r = await fetch(url, { signal: ctrl.signal, method: 'GET' });
        const latency_ms = Date.now() - t0;
        if (r.status >= 200 && r.status < 400) {
          return { status: 'running', latency_ms, error: null };
        }
        return { status: 'stopped', latency_ms, error: `HTTP ${r.status}` };
      } catch (err) {
        // ECONNREFUSED / abort / DNS / network — treat as 'stopped' (target not serving)
        // unless it's a config error (already handled above) → 'unknown'
        const msg = err?.name === 'AbortError' ? 'timeout' : (err?.message || 'unknown error');
        return { status: 'stopped', latency_ms: null, error: msg };
      } finally {
        clearTimeout(t);
      }
    }

    /**
     * TCP-connect probe with timeout. Connect-success = 'running'; ECONNREFUSED
     * or timeout = 'stopped'; bad host/port = 'unknown'.
     * @param {string} host
     * @param {number} port
     * @param {number} timeoutMs
     * @returns {Promise<{status: string, latency_ms: number | null, error: string | null}>}
     */
    export async function probeTcpPort(host, port, timeoutMs = 2000) {
      if (!host || !Number.isInteger(port) || port < 1 || port > 65535) {
        return { status: 'unknown', latency_ms: null, error: 'invalid host/port' };
      }
      const t0 = Date.now();
      return new Promise((resolve) => {
        const sock = new net.Socket();
        let settled = false;
        const settle = (status, error) => {
          if (settled) return;
          settled = true;
          sock.destroy();
          resolve({
            status,
            latency_ms: status === 'running' ? Date.now() - t0 : null,
            error
          });
        };
        sock.setTimeout(timeoutMs);
        sock.once('connect', () => settle('running', null));
        sock.once('timeout', () => settle('stopped', 'timeout'));
        sock.once('error', (err) => settle('stopped', err.code || err.message || 'connect error'));
        try {
          sock.connect(port, host);
        } catch (err) {
          settle('unknown', err.message || 'sync connect threw');
        }
      });
    }
    ```

| Boundary | Description |

| coordinator → external probes | Coordinator initiates outbound TCP / HTTP to localhost ports the rules JSON specifies. No untrusted input. |
| rules JSON → probe config | check_type, endpoint, port read from disk. Authored by us (not user input). |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-33-09-01 | D (DoS) | coordinator tick | mitigate | Probes run with bounded timeout (3s default); tick uses `Promise.all`, so slow probes don't serialize. Worst case: tick takes max(timeouts) ~3s, well under 5s tick interval — but if multiple probes time out, the tick CAN exceed 5s. Documented; revisit if SLA proves hard to meet. |
| T-33-09-02 | I (Info disclosure) | probeHttpHealth | accept | Probes hit ONLY localhost endpoints from a whitelisted rules file. No data exfiltration vector. |
| T-33-09-03 | T (Tampering) | rules JSON | accept | File on disk owned by user; same trust level as the source code. |
| T-33-09-04 | E (Elevation) | TCP probes | accept | Probes use `net.Socket.connect`; cannot mutate target service. Connect-only, no payload sent. |
| T-33-09-05 | S (Spoofing) | HTTP probes | accept | Localhost only; spoofing requires local code execution which already grants higher capability. |
| T-33-09-06 | R (Repudiation) | log entries | accept | Coordinator's `log()` already writes to .logs/health-coordinator.log with timestamps. Sufficient for debugging. |
