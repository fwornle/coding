NEW export (added to LLMSemanticMatcher.ts):

  /**
   * Typed error thrown by LLMSemanticMatcher.match() in `onError: 'throw'`
   * mode when the LLM response cannot be parsed even after the 5-stage
   * unwrap. Callers can `instanceof`-discriminate from generic LLM errors
   * (timeouts, network failures, etc.).
   *
   * The `cause` field carries the underlying JSON.parse SyntaxError; the
   * `raw` field carries the FIRST 1000 characters of the LLM's original
   * response (for diagnostic logging — full response intentionally
   * truncated to avoid leaking PII / huge payloads into error chains).
   *
   * Closes CR-03 (40-REVIEW.md).
   */
  export class LLMDedupParseError extends Error {
    readonly raw: string;
    readonly cause?: unknown;
    constructor(message: string, opts: { raw: string; cause?: unknown }) {
      super(message);
      this.name = 'LLMDedupParseError';
      this.raw = opts.raw.slice(0, 1000);
      if (opts.cause !== undefined) this.cause = opts.cause;
    }
  }

  // CR-03 + WR-08 fix (40-REVIEW.md). The unwrap is now a candidate-list-of-
  // tries: each stage CONTRIBUTES a candidate string when it can; the
  // function tries JSON.parse on each candidate in order. If ALL candidates
  // fail to parse, returns { matches: [] } (graceful no-match fallback —
  // callers in `onError: 'throw'` mode get an LLMDedupParseError thrown
  // upstream from match(), not from this function).
  //
  // SOURCE: 40-REVIEW.md offset 311-340 recipe (WR-08 fix).
  // PRIOR ART: 5-stage OKM unwrap port from Plan 40-04 — preserved across
  // the 4 response shapes (anchored, unanchored, bare-brace, raw).
  function parseDedupResponse(
    raw: string,
  ): { matches?: Array<{ newName: string; existingName: string }> } {
    const s = raw.trim();
    const candidates: string[] = [];

    // Stage 1: anchored fence (entire payload is ```json\n...\n```).
    const anchored = s.match(/^```(?:json)?\s*\n?([\s\S]*?)\n?\s*```\s*$/);
    if (anchored) candidates.push(anchored[1].trim());

    // Stage 2: unanchored fence (fenced block somewhere in the response).
    const unanchored = s.match(/```(?:json)?\s*\n([\s\S]*?)\n\s*```/);
    if (unanchored) candidates.push(unanchored[1].trim());

    // Stage 3: bare-brace extraction (prose-wrapped JSON).
    if (!s.startsWith('{') && !s.startsWith('[')) {
      const firstBrace = s.indexOf('{');
      const lastBrace = s.lastIndexOf('}');
      if (firstBrace >= 0 && lastBrace > firstBrace) {
        candidates.push(s.slice(firstBrace, lastBrace + 1));
      }
    }

    // Stage 4: raw (the unwrap-as-given path; covers the bare-JSON case).
    candidates.push(s);

    let lastError: unknown;
    for (const candidate of candidates) {
      try {
        return JSON.parse(candidate);
      } catch (err) {
        lastError = err;
        // try next candidate
      }
    }
    // All candidates failed. Return graceful no-match; match() surfaces
    // typed LLMDedupParseError when onError: 'throw' AND the raw response
    // contained at least one `{` (Contract A — see <behavior>).
    void lastError;
    return { matches: [] };
  }

      // is now a candidate-list-of-tries: each stage CONTRIBUTES a candidate
      // string when it can; we then try JSON.parse on each candidate in

      // PRIOR ART (Plan 40-04): The 5-stage OKM unwrap covered 4 response
      // shapes — anchored fence, unanchored fence, bare-brace, raw JSON.
      // The rewrite preserves all 4 by adding them as separate candidates
      // (Stage 4 'raw' = the input as-given, which covers the raw-JSON case).
      function parseDedupResponse(
        raw: string,
      ): { matches?: Array<{ newName: string; existingName: string }> } {
        const s = raw.trim();
        const candidates: string[] = [];

        // Stage 1: anchored fence.
        const anchored = s.match(/^```(?:json)?\s*\n?([\s\S]*?)\n?\s*```\s*$/);
        if (anchored) candidates.push(anchored[1].trim());

        // Stage 2: unanchored fence (LLM emitted prose + fenced block).
        const unanchored = s.match(/```(?:json)?\s*\n([\s\S]*?)\n\s*```/);
        if (unanchored) candidates.push(unanchored[1].trim());

        // Stage 3: bare-brace extraction (prose-wrapped JSON).
        if (!s.startsWith('{') && !s.startsWith('[')) {
          const firstBrace = s.indexOf('{');
          const lastBrace = s.lastIndexOf('}');
          if (firstBrace >= 0 && lastBrace > firstBrace) {
            candidates.push(s.slice(firstBrace, lastBrace + 1));
          }
        }

        // Stage 4: raw (covers the LLM-emits-pure-JSON case).
        candidates.push(s);

        for (const candidate of candidates) {
          try {
            return JSON.parse(candidate);
          } catch {
            // try next candidate
          }
        }
        // All candidates failed. Return graceful no-match — match() decides
        // whether to surface a typed LLMDedupParseError based on onError
        // AND the parsedTriedJson heuristic.
        return { matches: [] };
      }
