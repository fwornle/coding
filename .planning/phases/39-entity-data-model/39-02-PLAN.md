private mergeDescriptionSegment(

  newText: string,
  context: IngestionContext,
  timestamp: string,
): DescriptionSegment[] {

  if (match) {

}

    ```typescript
    // Phase 39 (DATA-02): per-segment provenance writer helper.
    // SOURCE: OKM `_work/.../ingestion/deduplicator.ts:380-416` lifted with 4 deltas:
    //   1. D-40 whitespace normalization (`text.trim().replace(/\s+/g, ' ')`) replaces OKM's `s.text === newText` exact equality.
    //   2. D-39 pure function — returns a NEW Entity, NEVER mutates input (OKM mutates in-place).
    //   3. Signature takes whole Entity (not segments array) so caller doesn't need to know the metadata key.
    //   4. D-41 segment-cap monitoring (stderr-warn at >100 segments OR >50 confirmations; no hard cap).
    import type { Entity, DescriptionSegment, SegmentConfirmation } from '../types/entity.js';

    const MAX_SEGMENTS_WARN = 100;        // D-41 monitoring threshold (no hard cap)
    const MAX_CONFIRMATIONS_WARN = 50;

    function normalize(text: string): string {
      return text.trim().replace(/\s+/g, ' ');
    }

    /**
     * Fold a new DescriptionSegment into the entity's metadata.descriptionSegments[].
     * Pure function: returns a new Entity; does NOT mutate input.
     *
     * Whitespace normalization (D-40): two segments with the SAME normalized text
     * are considered identical. Case-sensitive ('Code' !== 'code').
     *
     * If a normalized-identical existing segment is found:
     *   - Append a SegmentConfirmation {runId, provider, model, timestamp} to its confirmations[].
     * Otherwise:
     *   - Push the new segment (preserving caller-supplied confirmations[] or initializing to []).
     *
     * D-41: emits a process.stderr.write warning when the entity has >100 segments
     * or any segment has >50 confirmations. No hard cap; pruning policy deferred.
     */
    export function mergeDescriptionSegment(
      entity: Entity,
      newSegment: DescriptionSegment,
    ): Entity {
      const existingMetadata = entity.metadata ?? {};
      const existingSegments =
        ((existingMetadata as Record<string, unknown>).descriptionSegments as DescriptionSegment[] | undefined) ?? [];

      // D-39 deep clone — neither the segments array, the segment objects,
      // nor their confirmations arrays are reference-shared with the input.
      const segments: DescriptionSegment[] = existingSegments.map((s) => ({
        ...s,
        confirmations: [...(s.confirmations ?? [])],
      }));

      const normalizedNew = normalize(newSegment.text);
      const matchIndex = segments.findIndex((s) => normalize(s.text) === normalizedNew);

      if (matchIndex >= 0) {
        const match = segments[matchIndex];
        const confirmation: SegmentConfirmation = {
          runId: newSegment.runId,
          provider: newSegment.provider,
          model: newSegment.model,
          timestamp: newSegment.timestamp,
        };
        match.confirmations.push(confirmation);
        if (match.confirmations.length > MAX_CONFIRMATIONS_WARN) {
          process.stderr.write(
            `[km-core/segments] entity ${String(entity.id)} segment ${matchIndex} has ${match.confirmations.length} confirmations (>${MAX_CONFIRMATIONS_WARN}, monitoring per D-41)\n`,
          );
        }
      } else {
        segments.push({
          ...newSegment,
          confirmations: newSegment.confirmations ?? [],
        });
        if (segments.length > MAX_SEGMENTS_WARN) {
          process.stderr.write(
            `[km-core/segments] entity ${String(entity.id)} has ${segments.length} descriptionSegments (>${MAX_SEGMENTS_WARN}, monitoring per D-41)\n`,
          );
        }
      }

      return {
        ...entity,
        metadata: { ...existingMetadata, descriptionSegments: segments },
      };
    }
    ```
