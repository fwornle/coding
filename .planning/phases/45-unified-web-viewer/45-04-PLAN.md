<!-- Post-extension shape (this plan): -->
```typescript
// GET /ontology/classes — returns array of class NAME STRINGS (unchanged for back-compat)
// GET /ontology/classes?withDisplay=true — returns array of objects {name, level?, parent?, display?}
//   where display is sourced from .data/ontologies/{system}.display.json overlay.
async function getOntologyClasses(req, res) {
  const registry = req.app.locals.registry;
  if (req.query.withDisplay !== 'true') {
    res.json({ success: true, data: registry.classNames() });  // byte-identical to pre-Phase-45
    return;
  }
  const ontologyDir = req.app.locals.ontologyDir;
  const overlay = loadDisplayOverlay(ontologyDir, registry.domain);  // {} if file missing
  const data = registry.classNames().map(name => {
    const cls = registry.getClass(name);
    return { name, level: cls?.level, parent: cls?.parent, display: overlay[name] };
  });
  res.json({ success: true, data });
}
```
