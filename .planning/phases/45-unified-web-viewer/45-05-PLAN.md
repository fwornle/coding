```typescript
export type Pipeline = 'kpifw' | 'raas' | 'e2e';
export interface RcaDir { path: string; timestamp: string; findingCount: number; }
export type RcaDirGroups = { kpifw: RcaDir[]; raas: RcaDir[]; e2e: RcaDir[]; };
export interface IngestionStatus {
  active: boolean;
  pipeline?: Pipeline;
  dirPath?: string;
  currentStage?: 'extract' | 'dedup' | 'store' | 'synthesize' | 'resolve';
  progress?: number;  // 0-100
  startedAt?: string;
}
export interface PipelineEvent {
  type: 'connected' | 'stage' | 'progress' | 'complete' | 'error';
  stage?: string;
  progress?: number;
  message?: string;
  timestamp?: string;
}
export interface IngestStartResponse { success: boolean; runId?: string; error?: string; }

export class OkmRcaClient {
  constructor(private baseUrl: string) {}
  async listDirs(): Promise<RcaDirGroups>;
  async getStatus(): Promise<IngestionStatus>;
  async rcaIngest(pipeline: Pipeline, dirPath: string, opts?: { force?: boolean }): Promise<IngestStartResponse>;
  subscribeProgress(onMessage: (event: PipelineEvent) => void): EventSource;
}
```
