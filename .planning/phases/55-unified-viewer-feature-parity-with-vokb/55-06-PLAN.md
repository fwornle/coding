```typescript
// 1. GET /api/v1/stats
interface ViewerStats {
  nodeCount: number; edgeCount: number; evidenceCount: number;
  patternCount: number; orphanCount: number; componentCount: number;
  connectivity: number;          // 0..1
  lastUpdated: string;            // ISO
  activeSnapshot: { hash: string; message: string; date: string } | null
}
// Response envelope: { success: true, data: ViewerStats }

// 2. GET /api/v1/trends?top=20  (default top=20, max 100)
interface TrendingPattern {
  nodeId: string
  entity: { id: string; name: string; entityType: string; description?: string }
  trendScore: number
  trends: { last7Days: number; last30Days: number; last90Days: number }
}
// Response envelope: { success: true, data: { patterns: TrendingPattern[] } }

// 3. GET /api/v1/entities/:id/confidence
interface ConfidenceBreakdown {
  overall: number
  bands: { high: number; moderate: number; low: number }
  segments: Array<{ segmentId: string; confidence: number; source?: string }>
}
// Response envelope: { success: true, data: ConfidenceBreakdown }
// 404 envelope: { success: false, error: 'not_found' } — frontend falls back to client heuristic per UI-SPEC §16

// 4. GET /api/coding/observations/stream  (SSE)
// Content-Type: text/event-stream
// Each event: data: { ...Observation }\n\n  (Observation shape per ObservationWriter contract — snake_case session_id preserved)

// 5. GET /api/coding/lsl/sessions?since=<iso>&limit=200
interface LslSession {
  id: string                          // session sid
  startAt: string                     // ISO
  endAt: string | null                // null if currently running
  observationCount: number
  entityIds: string[]
}
// Response envelope: { success: true, data: { sessions: LslSession[] } }
```

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-55-06-01 | Denial of service | unbounded SSE subscribers leak memory | mitigate | `req.on('close')` unsubscribes; integration test asserts no-leak after disconnect |
