### Step 1: Sync stepPaused in fetchUKBStatusSuccess (ukbSlice.ts)

In the `fetchUKBStatusSuccess` reducer (line ~651-677), after merging processes, add top-level stepPaused sync:

```typescript
// After state.processes = [...mergedNewProcesses, ...existingRunningProcesses];

// Sync stepPaused from active process to top-level state
const activeProcess = state.processes.find(
  (p: UKBProcess) => p.status === 'running' || p.status === 'starting'
) as any;
if (activeProcess) {
  if (activeProcess.stepPaused !== undefined) {
    state.stepPaused = activeProcess.stepPaused === true;
    state.pausedAtStep = activeProcess.pausedAtStep || null;
  }
} else {
  // No active process — clear pause state
  state.stepPaused = false;
  state.pausedAtStep = null;
}
```

This ensures REST polling (the most reliable path) always syncs pause state to top-level.
