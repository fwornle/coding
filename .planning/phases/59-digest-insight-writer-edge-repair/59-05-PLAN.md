```javascript
#!/usr/bin/env node
import process from 'node:process';
import * as fsp from 'node:fs/promises';
import * as path from 'node:path';

const OBS_API = process.env.OBS_API_BASE || 'http://localhost:12436';
const ORPHAN_THRESHOLD = 10;          // SC#4: orphanCount <= 10
const SAMPLE_INTERVAL_MS = 60 * 60 * 1000;  // 1 hour
const TOTAL_SAMPLES = 24;
const LOG_DIR = path.resolve(process.cwd(), '.data');

function log(msg) { process.stderr.write(`[orphan-soak] ${msg}\n`); }

async function sampleStats() {
  const res = await fetch(`${OBS_API}/api/v1/stats`);
  if (!res.ok) throw new Error(`HTTP ${res.status} fetching stats`);
  const body = await res.json();
  const s = body.data;
  return {
    timestamp: new Date().toISOString(),
    orphanCount: s.orphanCount,
    nodeCount: s.nodeCount,
    connectivity: s.connectivity,
  };
}

async function main() {
  const samples = [];
  const sessionTs = new Date().toISOString().replace(/[:.]/g, '-');
  const outPath = path.join(LOG_DIR, `orphan-floor-soak-${sessionTs}.json`);
  let breached = false;
  for (let i = 0; i < TOTAL_SAMPLES; i++) {
    const sample = await sampleStats();
    samples.push(sample);
    log(`sample ${i+1}/${TOTAL_SAMPLES}: orphans=${sample.orphanCount} nodes=${sample.nodeCount}`);
    if (sample.orphanCount > ORPHAN_THRESHOLD) {
      breached = true;
      log(`THRESHOLD BREACH: orphans=${sample.orphanCount} > ${ORPHAN_THRESHOLD}`);
    }
    await fsp.writeFile(outPath, JSON.stringify({ samples, breached }, null, 2));
    if (i < TOTAL_SAMPLES - 1) {
      await new Promise(r => setTimeout(r, SAMPLE_INTERVAL_MS));
    }
  }
  log(`DONE: ${samples.length} samples written to ${outPath}`);
  process.exit(breached ? 1 : 0);
}

main().catch((e) => { log(`FATAL: ${e.stack}`); process.exit(1); });
```
The skeleton is a working starting point. Task 1 extends it with: (a) per-sample HTTP-failure handling (errors[] in the record, NO immediate fatal), (b) consecutive-failure escalation (3 in a row → exit code 1), (c) richer end-of-run summary including the max/min/median orphanCount across samples.

    **Feature 1 — per-sample HTTP-failure handling (non-fatal in isolation):**

    Wrap each `sampleStats()` call in try/catch:
    ```
    let sample
    try {
      sample = await sampleStats()
      consecutiveFailures = 0
    } catch (err) {
      consecutiveFailures++
      sample = {
        timestamp: new Date().toISOString(),
        orphanCount: -1,                // sentinel for "no data"
        nodeCount: -1,
        connectivity: -1,
        errors: [err.message],
      }
      log(`sample ${i+1}/${TOTAL_SAMPLES}: FETCH FAILED (${err.message}); consecutive=${consecutiveFailures}`)
    }
    ```

    A single-sample failure does NOT set `breached` (because orphanCount: -1 is not a real measurement, NOT a true breach). The threshold check `if (sample.orphanCount > ORPHAN_THRESHOLD)` skips the -1 sentinel since `-1 > 10` is false.

    **Feature 2 — consecutive-failure escalation:**

    Top of `main()`: `let consecutiveFailures = 0; const CONSECUTIVE_FAILURE_LIMIT = 3;`

    After each sample: if `consecutiveFailures >= CONSECUTIVE_FAILURE_LIMIT`, the obs-api has been down for >3 hours and the soak data is invalid. Log a fatal message, write the partial session log, exit code 1:
    ```
    if (consecutiveFailures >= CONSECUTIVE_FAILURE_LIMIT) {
      log(`FATAL: ${consecutiveFailures} consecutive sample failures — obs-api appears unreachable; aborting soak`)
      await fsp.writeFile(outPath, JSON.stringify({ samples, breached, aborted: true, reason: 'consecutive-failures' }, null, 2))
      process.exit(1)
    }
    ```

    **Feature 3 — richer end-of-run summary (computed from samples, written to outPath + printed to stdout):**

    After the loop closes:
    ```
    const validSamples = samples.filter(s => s.orphanCount >= 0)
    const orphanCounts = validSamples.map(s => s.orphanCount)
    const summary = {
      totalSamples: samples.length,
      validSamples: validSamples.length,
      failedSamples: samples.length - validSamples.length,
      threshold: ORPHAN_THRESHOLD,
      breached,
      orphanCount: {
        max: orphanCounts.length ? Math.max(...orphanCounts) : null,
        min: orphanCounts.length ? Math.min(...orphanCounts) : null,
        mean: orphanCounts.length ? Math.round(orphanCounts.reduce((a,b)=>a+b,0) / orphanCounts.length) : null,
      },
      startedAt: samples[0]?.timestamp,
      endedAt: samples[samples.length - 1]?.timestamp,
      sessionLogPath: outPath,
    }
    await fsp.writeFile(outPath, JSON.stringify({ samples, breached, summary }, null, 2))
    process.stdout.write(JSON.stringify(summary, null, 2) + '\n')
    log(`DONE: ${samples.length} samples written to ${outPath}`)
    process.exit(breached ? 1 : 0)
    ```

    **Pre-flight gate:** before the loop, do ONE sample fetch with NO try/catch. If it throws, the obs-api is not reachable AT ALL — exit code 2 (pre-flight failure, distinct from in-flight failures):
    ```
    try {
      const probe = await sampleStats()
      log(`pre-flight OK: orphans=${probe.orphanCount} nodes=${probe.nodeCount}`)
    } catch (err) {
      log(`FATAL: pre-flight fetch failed: ${err.message}`)
      process.exit(2)
    }
    ```
    The pre-flight fetch is BEFORE the loop and BEFORE outPath has been created — exit-on-failure leaves no partial session log (correct shape; pre-flight failure means the soak never started).

Constants are top-of-file: `ORPHAN_THRESHOLD=10`, `SAMPLE_INTERVAL_MS=3600000`, `TOTAL_SAMPLES=24`, `CONSECUTIVE_FAILURE_LIMIT=3`, `OBS_API` from env.
