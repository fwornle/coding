<!-- Installer script (mirrors scripts/install-lsl-resolver-launchd.sh verbatim) -->

scripts/install-sub-agent-launchd.sh iterates 4 plists:
```bash
#!/bin/bash
set -euo pipefail
PLISTS=(
  com.coding.sub-agent-sweep
  com.coding.sub-agent-live-claude
  com.coding.sub-agent-live-opencode
  com.coding.sub-agent-live-copilot
)
LAUNCH_AGENTS=$HOME/Library/LaunchAgents
mkdir -p "$LAUNCH_AGENTS" "$PWD/.logs"
for LABEL in "${PLISTS[@]}"; do
  SRC="$PWD/launchd/${LABEL}.plist"
  DST="$LAUNCH_AGENTS/${LABEL}.plist"
  if ! diff -q "$SRC" "$DST" >/dev/null 2>&1; then
    cp "$SRC" "$DST"
    echo "[install-sub-agent] installed plist at $DST" >&2
  fi
  launchctl bootout "gui/$UID/$LABEL" 2>/dev/null || true
  launchctl bootstrap "gui/$UID" "$DST"
  if launchctl list | grep -qF "$LABEL"; then
    echo "[install-sub-agent] OK: $LABEL is loaded" >&2
  else
    echo "[install-sub-agent] FAIL: $LABEL did not load" >&2
    exit 1
  fi
done
echo "[install-sub-agent] all 4 jobs installed" >&2
```
