1. Calls the shared retrieval client with a generic project query and context
2. Writes the returned markdown to the agent's native context file
3. Exits silently on any error (fail-open)

**Common pattern for all three adapters:**

```javascript
#!/usr/bin/env node
/**
 * Knowledge Injection Adapter for {AgentName} (session-start)
 * Called once at session start by launch-agent-common.sh.
 * Writes retrieved knowledge to {target_file} for {AgentName} to load as context.
 */
import { callRetrieval } from './retrieval-client.js';
import { execSync } from 'node:child_process';
import { writeFileSync, mkdirSync } from 'node:fs';
import { dirname, basename } from 'node:path';

const cwd = process.env.CODING_PROJECT_DIR || process.env.TARGET_PROJECT_DIR || process.cwd();
const project = basename(cwd);

async function main() {
  try {
    // Gather recent files from git for context (D-08)
    let recentFiles = [];
    try {
      const output = execSync('git diff --name-only HEAD~3 2>/dev/null', { cwd, encoding: 'utf8', timeout: 3000 });
      recentFiles = output.trim().split('\n').filter(Boolean).slice(0, 20);
    } catch { /* no git or no commits -- skip */ }

    const result = await callRetrieval({
      query: `project context for ${project}`,
      budget: 1000,
      threshold: 0.65,  // slightly lower for session-start generic queries (D-11)
      context: { project, cwd, recent_files: recentFiles },
      timeout: 3000,     // slightly longer timeout for session-start (not per-prompt)
    });

    if (!result || !result.markdown || result.meta?.results_count === 0) {
      return; // No relevant knowledge -- skip writing
    }

    // Write context file
    const targetFile = `${cwd}/{AGENT_CONTEXT_PATH}`;
    mkdirSync(dirname(targetFile), { recursive: true });

    const header = `<!-- Auto-generated by coding knowledge injection - ${new Date().toISOString()} -->\n`;
    const content = header + `# Knowledge Context\n\n${result.markdown}\n`;
    writeFileSync(targetFile, content, 'utf8');

    process.stderr.write(`[knowledge-{agent}] Wrote ${content.length} chars to ${targetFile}\n`);
  } catch (err) {
    process.stderr.write(`[knowledge-{agent}] Error: ${err.message}\n`);
    // Fail-open: exit 0 regardless
  }
}

main().then(() => process.exit(0));
```

```bash
# Inject knowledge context for non-Claude agents (D-06)
# Claude uses a per-prompt UserPromptSubmit hook (registered globally).
# Other agents get a session-start context file written before launch.
_inject_knowledge_context() {
  local agent="$AGENT_NAME"
  local hooks_dir="$CODING_REPO/src/hooks"

  # Claude uses per-prompt hook (registered in ~/.claude/settings.json) -- skip here
  if [ "$agent" = "claude" ]; then
    return 0
  fi

  local adapter="$hooks_dir/knowledge-injection-${agent}.js"
  if [ ! -f "$adapter" ]; then
    _agent_log "No knowledge injection adapter for ${agent} -- skipping"
    return 0
  fi

  _agent_log "Injecting knowledge context for ${AGENT_DISPLAY_NAME}..."

  # Export target project dir for the adapter to use
  export TARGET_PROJECT_DIR="${TARGET_PROJECT_DIR}"
  export CODING_PROJECT_DIR="${TARGET_PROJECT_DIR}"

  # Run adapter with timeout (fail-open -- never block agent startup)
  if timeout 10 node "$adapter" 2>/dev/null; then
    _agent_log "Knowledge context injected for ${AGENT_DISPLAY_NAME}"
  else
    _agent_log "Knowledge injection skipped (service unavailable or timeout)"
  fi
}
```
